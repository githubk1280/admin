
CREATE TABLE `privilege` (
  `PRIVILEGE_ID` int(11) NOT NULL AUTO_INCREMENT,
  `PRIVILEGE_CODE` varchar(45) DEFAULT NULL,
  `DESCRIPTION` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`PRIVILEGE_ID`)
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8;


CREATE TABLE `project` (
  `PROJECT_ID` int(11) NOT NULL AUTO_INCREMENT,
  `NEED_PRINCPAL_ASSIST` char(2) NOT NULL DEFAULT 'N' COMMENT '是否需要主管协助',
  `PRIORITY` int(11) DEFAULT NULL,
  `START_DATE` date DEFAULT NULL,
  `END_DATE` date DEFAULT NULL,
  `PROJECT_CONTENT` blob,
  `POTENTIAL_EXPENDITURE` varchar(50) DEFAULT NULL COMMENT '项目潜在经费',
  `SAMPLE_COUNT` int(11) DEFAULT NULL COMMENT '样本数量',
  `RESEARCH_DIRECTION` varchar(90) DEFAULT NULL COMMENT '研究方向',
  `TECHNOLOGY_PLATFORM_DEMAND` varchar(255) DEFAULT NULL COMMENT '技术平台要求',
  `HAS_CORPORATED_INTENTION` char(2) DEFAULT NULL COMMENT '是否有意10X1000合作',
  `NEED_PROJECT_PLAN` char(2) DEFAULT NULL COMMENT '课题方案是否需要',
  `NEED_SCI` char(2) DEFAULT NULL COMMENT '需要发SCI文章',
  `PROJECT_NOTE` blob COMMENT '项目备注',
  `TYPE_TRANSFER_INFO` int(11) DEFAULT NULL,
  `PROJECT_NAME` varchar(45) DEFAULT NULL,
  `PROJECT_TYPE` varchar(45) DEFAULT NULL COMMENT '项目类型',
  PRIMARY KEY (`PROJECT_ID`)
) ENGINE=InnoDB AUTO_INCREMENT=5 DEFAULT CHARSET=utf8;



CREATE TABLE `project_employee` (
  `PROJECT_ID` int(11) NOT NULL,
  `EMPLOYEE_ID` int(11) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8;


CREATE TABLE `employee` (
  `EMPLOYEE_ID` int(11) NOT NULL AUTO_INCREMENT,
  `EMPLOYEE_NAME` varchar(96) NOT NULL,
  `FK_EMPLOYEE_ROLE_ID` int(11) NOT NULL,
  `FK_PRIVILEGE_ID` int(11) DEFAULT NULL,
  `PASSWORD` varchar(96) DEFAULT NULL,
  PRIMARY KEY (`EMPLOYEE_ID`)
) ENGINE=InnoDB AUTO_INCREMENT=4 DEFAULT CHARSET=utf8;

CREATE TABLE `employee_role` (
  `EMPLOYEE_ROLE_ID` int(11) NOT NULL AUTO_INCREMENT,
  `EMPLOYEE_ROLE` char(2) NOT NULL,
  `DESCRIPTION` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`EMPLOYEE_ROLE_ID`)
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8;


CREATE TABLE `out_source` (
  `OUT_SOURCE_ID` int(11) NOT NULL AUTO_INCREMENT,
  `OUT_SOURCE_UNIT` varchar(125) DEFAULT NULL,
  `ADRESS` varchar(255) DEFAULT NULL,
  `FK_CUSTOMER_ID` int(11) DEFAULT NULL,
  `SEND_SAMPLE_DATE` date DEFAULT NULL,
  `SEND_SAMPLE_COUNT` int(11) DEFAULT NULL,
  `IS_EXPERIMENT_RESULT_RECIVED` char(2) DEFAULT NULL,
  `OUT_SOURCE_STATUS` varchar(45) DEFAULT NULL,
  `FK_PROJECT_ID` int(11) DEFAULT NULL,
  `OUT_SOURCE_TYPE` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`OUT_SOURCE_ID`)
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8;


CREATE TABLE `customers` (
  `CUSTOMER_ID` int(11) NOT NULL,
  `CUSTOMER_NAME` varchar(128) NOT NULL,
  `SEX` char(2) NOT NULL,
  `AGE` int(5) DEFAULT NULL,
  `PHONE` varchar(45) DEFAULT NULL,
  `CELLPHONE` varchar(45) DEFAULT NULL,
  `WORK_UNIT` varchar(45) DEFAULT NULL,
  `DEPARTMENT` varchar(45) DEFAULT NULL,
  `POSITION` varchar(45) DEFAULT NULL,
  `EMAIL` varchar(45) DEFAULT NULL,
  `ADRESS` varchar(45) DEFAULT NULL,
  `RESEARCH_DIRECTION` varchar(45) DEFAULT NULL,
  `RESPONSIBLE_AREA` varchar(45) DEFAULT NULL,
  `CUSTOMER_LEVEL` int(11) DEFAULT '1',
  `PRINCIPAL_ID` int(11) DEFAULT NULL,
  `PROJECT_ID` int(11) DEFAULT NULL,
  `RELATIONSHIP` varchar(45) DEFAULT NULL,
  `PRINCIPAL_NUM` int(11) DEFAULT NULL,
  PRIMARY KEY (`CUSTOMER_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

CREATE TABLE `project_progress` (
  `PROJECT_PROGRESS_ID` int(11) NOT NULL AUTO_INCREMENT,
  `EMPLOYEE_NAME` varchar(25) NOT NULL,
  `PROGRESS_CONTENT` varchar(255) DEFAULT NULL,
  `PROGRESS_DATE` date DEFAULT NULL,
  `PROJECT_ID` int(11) DEFAULT NULL,
  PRIMARY KEY (`PROJECT_PROGRESS_ID`)
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8;
