CREATE TABLE `employee` (
  `EMPLOYEE_ID` int(11) NOT NULL AUTO_INCREMENT,
  `EMPLOYEE_NAME` varchar(45) NOT NULL,
  `FK_EMPLOYEE_ROLE_ID` int(11) NOT NULL,
  `FK_PRIVILEGE_ID` int(11) DEFAULT NULL,
  `PASSWORD` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`EMPLOYEE_ID`),
  KEY `FK_EMP_PRIVILEGE_idx` (`FK_PRIVILEGE_ID`),
  KEY `FK_EMPLOYEE_ROLE_idx` (`FK_EMPLOYEE_ROLE_ID`),
  CONSTRAINT `FK_EMPLOYEE_ROLE` FOREIGN KEY (`FK_EMPLOYEE_ROLE_ID`) REFERENCES `employee_role` (`EMPLOYEE_ROLE_ID`) ON DELETE NO ACTION ON UPDATE NO ACTION,
  CONSTRAINT `FK_EMP_PRIVILEGE` FOREIGN KEY (`FK_PRIVILEGE_ID`) REFERENCES `privilege` (`PRIVILEGE_ID`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB AUTO_INCREMENT=4 DEFAULT CHARSET=utf8;


CREATE TABLE `employee_role` (
  `EMPLOYEE_ROLE_ID` int(11) NOT NULL AUTO_INCREMENT,
  `EMPLOYEE_ROLE` char(2) NOT NULL,
  `DESCRIPTION` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`EMPLOYEE_ROLE_ID`)
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8;


CREATE TABLE `privilege` (
  `PRIVILEGE_ID` int(11) NOT NULL AUTO_INCREMENT,
  `PRIVILEGE_CODE` varchar(45) DEFAULT NULL,
  `DESCRIPTION` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`PRIVILEGE_ID`)
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8;


CREATE TABLE `project` (
  `PROJECT_ID` int(11) NOT NULL AUTO_INCREMENT,
  `NEED_PRINCPAL_ASSIST` char(2) NOT NULL DEFAULT 'N',
  `PRIORITY` int(11) DEFAULT NULL,
  `START_DATE` date DEFAULT NULL,
  `END_DATE` date DEFAULT NULL,
  `PROJECT_CONTENT` blob,
  `POTENTIAL_EXPENDITURE` varchar(50) DEFAULT NULL,
  `SAMPLE_COUNT` int(11) DEFAULT NULL,
  `RESEARCH_DIRECTION` varchar(90) DEFAULT NULL,
  `TECHNOLOGY_PLATFORM_DEMAND` varchar(255) DEFAULT NULL,
  `HAS_CORPORATED_INTENTION` char(2) DEFAULT NULL,
  `NEED_PROJECT_PLAN` char(2) DEFAULT NULL,
  `NEED_SCI` char(2) DEFAULT NULL,
  `PROJECT_NOTE` blob,
  `TYPE_TRANSFER_INFO` int(11) DEFAULT NULL,
  `PROJECT_NAME` varchar(45) DEFAULT NULL,
  `FK_OS_ID` int(11) DEFAULT NULL,
  PRIMARY KEY (`PROJECT_ID`),
  KEY `FK_PRO_OS_idx` (`FK_OS_ID`),
  CONSTRAINT `FK_PRO_OS` FOREIGN KEY (`FK_OS_ID`) REFERENCES `out_source` (`OUT_SOURCE_ID`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB AUTO_INCREMENT=5 DEFAULT CHARSET=utf8;



CREATE TABLE `project_employee` (
  `PROJECT_ID` int(11) NOT NULL,
  `EMPLOYEE_ID` int(11) NOT NULL,
  KEY `FK_EMP_idx` (`EMPLOYEE_ID`),
  KEY `FK_PRO_idx` (`PROJECT_ID`),
  CONSTRAINT `FK_PRO` FOREIGN KEY (`PROJECT_ID`) REFERENCES `project` (`PROJECT_ID`) ON DELETE NO ACTION ON UPDATE NO ACTION,
  CONSTRAINT `FK_EMP` FOREIGN KEY (`EMPLOYEE_ID`) REFERENCES `employee` (`EMPLOYEE_ID`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB DEFAULT CHARSET=utf8;


CREATE TABLE `out_source` (
  `OUT_SOURCE_ID` int(11) NOT NULL AUTO_INCREMENT,
  `OUT_SOURCE_UNIT` varchar(125) DEFAULT NULL,
  `ADRESS` varchar(255) DEFAULT NULL,
  `FK_CUSTOMER_ID` int(11) DEFAULT NULL,
  `SEND_SAMPLE_DATE` date DEFAULT NULL,
  `SEND_SAMPLE_COUNT` int(11) DEFAULT NULL,
  `IS_EXPERIMENT_RESULT_RECIVED` char(2) DEFAULT NULL,
  `OUT_SOURCE_STATUS` varchar(45) DEFAULT NULL,
  `FK_PROJECT_ID` int(11) DEFAULT NULL,
  `OUT_SOURCE_TYPE` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`OUT_SOURCE_ID`),
  KEY `FK_OS_SUSTOMER_idx` (`FK_CUSTOMER_ID`),
  KEY `FK_OS_PROJECT_idx` (`FK_PROJECT_ID`),
  CONSTRAINT `FK_OS_PROJECT` FOREIGN KEY (`FK_PROJECT_ID`) REFERENCES `project` (`PROJECT_ID`) ON DELETE NO ACTION ON UPDATE NO ACTION,
  CONSTRAINT `FK_OS_SUSTOMER` FOREIGN KEY (`FK_CUSTOMER_ID`) REFERENCES `customer` (`PK_CUSTOMER_ID`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8;
